Building a Knowledge Base with Claude: Chunking, Embedding, and Retrieval

A complete Python implementation — document ingestion, chunking strategies, Voyage AI embeddings, pgvector storage, hybrid search, reranking, and a full RAG pipeline that assembles context for Claude.

Apr 14, 2026  ·  13 min read  ·  CEO Editorial

RAG (Retrieval-Augmented Generation) is the standard pattern for giving Claude access to a private knowledge base without fine-tuning. The concept is simple; the implementation details matter. Wrong chunk size, poor embedding model choice, or naive retrieval all produce answers that hallucinate or miss relevant context.

This guide walks through a production-grade knowledge base implementation: chunking strategy, Voyage AI embeddings (Anthropic's recommended provider), pgvector for storage and search, hybrid keyword + semantic retrieval, and the final prompt assembly step that feeds context to Claude.

Document Chunking

Chunk size is the most impactful parameter in a RAG pipeline. Too large: irrelevant content dilutes the answer. Too small: context is fragmented and lacks coherence.

from dataclasses import dataclass, field
from typing import Iterator
import re


@dataclass
class Chunk:
    text: str
    doc_id: str
    chunk_index: int
    metadata: dict = field(default_factory=dict)

    @property
    def token_estimate(self) -> int:
        # ~4 chars per token is a reliable estimate for English
        return len(self.text) // 4


class RecursiveChunker:
    """
    Split documents recursively by separators, targeting a token size.
    Priority order: paragraphs → sentences → words.
    Maintains overlap to preserve cross-chunk context.
    """

    SEPARATORS = ["\n\n", "\n", ". ", " "]

    def __init__(self, chunk_size: int = 512, overlap: int = 64):
        self.chunk_size = chunk_size  # tokens
        self.overlap = overlap        # tokens of overlap between chunks
        self._char_limit = chunk_size * 4
        self._overlap_chars = overlap * 4

    def _split_text(self, text: str, sep_idx: int = 0) -> list[str]:
        if len(text) <= self._char_limit or sep_idx >= len(self.SEPARATORS):
            return [text]

        sep = self.SEPARATORS[sep_idx]
        parts = text.split(sep)
        chunks, current = [], ""

        for part in parts:
            candidate = current + sep + part if current else part
            if len(candidate) <= self._char_limit:
                current = candidate
            else:
                if current:
                    chunks.append(current)
                # Part itself too large — recurse with next separator
                if len(part) > self._char_limit:
                    chunks.extend(self._split_text(part, sep_idx + 1))
                    current = ""
                else:
                    current = part

        if current:
            chunks.append(current)
        return chunks

    def chunk(self, text: str, doc_id: str, metadata: dict = None) -> list[Chunk]:
        raw_chunks = self._split_text(text.strip())
        result = []
        for i, chunk_text in enumerate(raw_chunks):
            # Add overlap: prepend tail of previous chunk
            if i > 0 and self.overlap > 0:
                prev_tail = raw_chunks[i - 1][-self._overlap_chars:]
                chunk_text = prev_tail + " " + chunk_text

            result.append(Chunk(
                text=chunk_text.strip(),
                doc_id=doc_id,
                chunk_index=i,
                metadata=metadata or {},
            ))
        return result
Chunk Size Guidelines 512 tokens works well for Q&A over documentation. Use 256 for precise factual retrieval (API docs, specs). Use 1024 for narrative content where paragraph-level coherence matters. The overlap (10–15% of chunk size) prevents answers that straddle a chunk boundary from being missed.

Embeddings with Voyage AI

Anthropic recommends Voyage AI for embeddings used with Claude. Their voyage-3 model produces 1024-dimensional vectors and significantly outperforms OpenAI's ada-002 on retrieval benchmarks.

import voyageai
from tenacity import retry, stop_after_attempt, wait_exponential
import asyncio


voyage_client = voyageai.AsyncClient()  # uses VOYAGE_API_KEY env var


@retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=2, max=10))
async def embed_batch(texts: list[str], input_type: str = "document") -> list[list[float]]:
    """
    Embed a batch of texts.
    input_type: "document" for indexing, "query" for search queries.
    Voyage uses different internal representations for each.
    """
    result = await voyage_client.embed(
        texts,
        model="voyage-3",
        input_type=input_type,
    )
    return result.embeddings


async def embed_chunks(chunks: list[Chunk], batch_size: int = 128) -> list[list[float]]:
    """Embed all chunks in batches."""
    all_embeddings = []
    texts = [c.text for c in chunks]

    for i in range(0, len(texts), batch_size):
        batch = texts[i:i + batch_size]
        embeddings = await embed_batch(batch, input_type="document")
        all_embeddings.extend(embeddings)

    return all_embeddings

pgvector Storage

Store chunks and their embeddings in PostgreSQL with the pgvector extension. Supports both exact and approximate nearest-neighbor search with HNSW indexing.

import asyncpg
from pgvector.asyncpg import register_vector
import json


async def setup_schema(conn: asyncpg.Connection):
    await conn.execute("CREATE EXTENSION IF NOT EXISTS vector")
    await conn.execute("""
        CREATE TABLE IF NOT EXISTS chunks (
            id          BIGSERIAL PRIMARY KEY,
            doc_id      TEXT NOT NULL,
            chunk_index INT NOT NULL,
            text        TEXT NOT NULL,
            metadata    JSONB DEFAULT '{}',
            embedding   vector(1024),
            created_at  TIMESTAMPTZ DEFAULT NOW()
        )
    """)
    # HNSW index — fast approximate search at scale
    await conn.execute("""
        CREATE INDEX IF NOT EXISTS chunks_embedding_idx
        ON chunks USING hnsw (embedding vector_cosine_ops)
        WITH (m = 16, ef_construction = 64)
    """)
    # Full-text search index for hybrid retrieval
    await conn.execute("""
        CREATE INDEX IF NOT EXISTS chunks_fts_idx
        ON chunks USING gin (to_tsvector('english', text))
    """)


async def upsert_chunks(
    conn: asyncpg.Connection,
    chunks: list[Chunk],
    embeddings: list[list[float]],
):
    await register_vector(conn)
    # Delete existing chunks for this doc before re-inserting
    doc_ids = list({c.doc_id for c in chunks})
    await conn.execute("DELETE FROM chunks WHERE doc_id = ANY($1)", doc_ids)

    await conn.executemany(
        """
        INSERT INTO chunks (doc_id, chunk_index, text, metadata, embedding)
        VALUES ($1, $2, $3, $4, $5)
        """,
        [
            (c.doc_id, c.chunk_index, c.text, json.dumps(c.metadata), emb)
            for c, emb in zip(chunks, embeddings)
        ],
    )

Hybrid Retrieval

Pure vector search misses exact keyword matches (product names, error codes, version numbers). Pure keyword search misses semantic similarity. Hybrid search — combining both with Reciprocal Rank Fusion — outperforms either alone.

from dataclasses import dataclass


@dataclass
class RetrievedChunk:
    chunk_id: int
    doc_id: str
    text: str
    metadata: dict
    vector_rank: int | None = None
    keyword_rank: int | None = None
    rrf_score: float = 0.0


async def hybrid_search(
    conn: asyncpg.Connection,
    query: str,
    query_embedding: list[float],
    top_k: int = 20,
    rrf_k: int = 60,
) -> list[RetrievedChunk]:
    await register_vector(conn)

    # 1. Vector search — cosine similarity
    vector_rows = await conn.fetch(
        """
        SELECT id, doc_id, text, metadata,
               1 - (embedding <=> $1::vector) AS score
        FROM chunks
        ORDER BY embedding <=> $1::vector
        LIMIT $2
        """,
        query_embedding, top_k,
    )

    # 2. Full-text keyword search
    keyword_rows = await conn.fetch(
        """
        SELECT id, doc_id, text, metadata,
               ts_rank(to_tsvector('english', text), plainto_tsquery('english', $1)) AS score
        FROM chunks
        WHERE to_tsvector('english', text) @@ plainto_tsquery('english', $1)
        ORDER BY score DESC
        LIMIT $2
        """,
        query, top_k,
    )

    # 3. Reciprocal Rank Fusion
    chunks: dict[int, RetrievedChunk] = {}

    for rank, row in enumerate(vector_rows, start=1):
        chunks[row["id"]] = RetrievedChunk(
            chunk_id=row["id"], doc_id=row["doc_id"],
            text=row["text"], metadata=dict(row["metadata"]),
            vector_rank=rank,
        )
        chunks[row["id"]].rrf_score += 1 / (rrf_k + rank)

    for rank, row in enumerate(keyword_rows, start=1):
        if row["id"] not in chunks:
            chunks[row["id"]] = RetrievedChunk(
                chunk_id=row["id"], doc_id=row["doc_id"],
                text=row["text"], metadata=dict(row["metadata"]),
            )
        chunks[row["id"]].keyword_rank = rank
        chunks[row["id"]].rrf_score += 1 / (rrf_k + rank)

    return sorted(chunks.values(), key=lambda c: c.rrf_score, reverse=True)[:top_k]

Reranking

Retrieval returns candidates; reranking picks the best ones. Voyage AI's rerank-2 is a cross-encoder that scores each (query, chunk) pair — much more accurate than embedding similarity alone for final selection.

async def rerank(
    query: str,
    candidates: list[RetrievedChunk],
    top_n: int = 5,
) -> list[RetrievedChunk]:
    """Rerank candidates with Voyage rerank-2, return top_n."""
    if not candidates:
        return []

    result = await voyage_client.rerank(
        query=query,
        documents=[c.text for c in candidates],
        model="rerank-2",
        top_k=top_n,
    )

    # Map reranked indices back to original chunks
    return [candidates[r.index] for r in result.results]

The Complete RAG Pipeline

from anthropic import AsyncAnthropic

claude = AsyncAnthropic()

SYSTEM_PROMPT = """You are a knowledgeable assistant. Answer questions using only the provided context.
If the answer is not in the context, say "I don't have that information in my knowledge base."
Always cite which document section your answer comes from."""


async def rag_query(
    conn: asyncpg.Connection,
    question: str,
    top_k: int = 20,
    rerank_n: int = 5,
) -> str:
    # 1. Embed the query (use "query" input_type, not "document")
    query_emb = (await embed_batch([question], input_type="query"))[0]

    # 2. Hybrid retrieval
    candidates = await hybrid_search(conn, question, query_emb, top_k=top_k)

    # 3. Rerank to top N
    top_chunks = await rerank(question, candidates, top_n=rerank_n)

    # 4. Assemble context block
    context_parts = []
    for i, chunk in enumerate(top_chunks, start=1):
        source = chunk.metadata.get("title", chunk.doc_id)
        context_parts.append(f"[{i}] Source: {source}\n{chunk.text}")
    context = "\n\n---\n\n".join(context_parts)

    # 5. Call Claude with assembled context
    msg = await claude.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=2048,
        system=SYSTEM_PROMPT,
        messages=[{
            "role": "user",
            "content": f"Context:\n\n{context}\n\nQuestion: {question}",
        }],
    )
    return msg.content[0].text


# Full ingestion pipeline
async def ingest_document(
    conn: asyncpg.Connection,
    text: str,
    doc_id: str,
    metadata: dict = None,
    chunk_size: int = 512,
):
    chunker = RecursiveChunker(chunk_size=chunk_size, overlap=64)
    chunks = chunker.chunk(text, doc_id=doc_id, metadata=metadata)
    embeddings = await embed_chunks(chunks)
    await upsert_chunks(conn, chunks, embeddings)
    return len(chunks)
Production Tips Cache query embeddings by hashing the query string — identical queries don't need re-embedding. For frequently asked questions, cache the full RAG response with a short TTL (5–10 minutes). Use EXPLAIN ANALYZE on your vector queries to verify the HNSW index is being used — a sequential scan on a 100K-chunk table will be noticeably slow.

Summary

  • Chunk with overlap — 10–15% overlap prevents answers straddling chunk boundaries from being missed
  • Use voyage-3 with correct input_type — "document" at index time, "query" at search time; mixing them degrades recall
  • Hybrid search beats either alone — RRF fusion handles both semantic and exact-match queries
  • Rerank before generating — cross-encoder reranking on the top-20 candidates to select the best 5 consistently improves answer quality
  • HNSW index on pgvector — sub-millisecond ANN search; tune ef_construction up for higher recall if you have the build time
  • Cache aggressively — query embeddings, common query responses, and chunk data are all safe to cache
